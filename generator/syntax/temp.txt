#include "token.h"
#include "grammar.h"
#include "syntax_rule_parser.h"
#include "lr_parser.h"
#include <filesystem>
#include <iostream>


int main()
{
    std::filesystem::path file{"./resource/rule/syntax_rule.txt"};

    Grammar grammar;
    SyntaxRuleParser::parse(file, grammar);
    // grammar.show();

    grammar.compute_first_sets();
    grammar.compute_follow_sets();
    // grammar.show_first_follow();

    LRParser parser{&grammar};
    parser.build_states();
    parser.construct_tables();
    // parser.show_states();
    // parser.merge_states();


    std::vector<std::vector<Token>> tokens_list {
        {{"INT", "int"}, {"IDENTIFIER", "x"}, {"SEMICOLON", ";"}},
        {{"INT", "int"}, {"IDENTIFIER", "y"}, {"LPAREN", "("}, {"RPAREN", ")"}, {"LBRACE", "{"}, {"RBRACE", "}"}},
        {{"INT", "int"}, {"IDENTIFIER", "z"}, {"LPAREN", "("}, {"RPAREN", ")"}, {"LBRACE", "{"}, {"RETURN", "return"}, {"NUMBER", "0"}, {"SEMICOLON", ";"}, {"RBRACE", "}"}},
        {{"INT", "int"}, {"IDENTIFIER", "a"}, {"LPAREN", "("}, {"RPAREN", ")"}, {"LBRACE", "{"}, {"IDENTIFIER", "b"}, {"ASSIGN", "="}, {"NUMBER", "5"}, {"SEMICOLON", ";"}, {"RBRACE", "}"}},
        {{"INT", "int"}, {"IDENTIFIER", "c"}, {"LPAREN", "("}, {"RPAREN", ")"}, {"LBRACE", "{"}, {"IF", "if"}, {"LPAREN", "("}, {"IDENTIFIER", "d"}, {"RPAREN", ")"}, {"RETURN", "return"}, {"NUMBER", "10"}, {"SEMICOLON", ";"}, {"RBRACE", "}"}},
    };

    int counter = 0;
    for (auto& tokens : tokens_list)
    {
        std::cout << counter << ": ";
        std::cout << (parser.parse(tokens) ? "Passed" : "Failed") << '\n';
        const ASTTree& ast_tree = parser.get_ast_tree();
        ast_tree.show();
        ++counter;
    }

    return 0;
}
